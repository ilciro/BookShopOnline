				+ "`descrizione`,"
				+ "`dataPubblicazione`,"
				+ "`copieRimanenti`,"
				+ "`disp`,"
				+ "`prezzo`)"
				+ "VALUES"
				+ "(?,?,?,?,?,?,?,?,?,?)";

			PreparedStatement prepQ=conn.prepareStatement(query))
		{

			prepQ.setString(1,r.getTitolo());
